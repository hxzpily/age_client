	 var src = {fieldA: null, fieldB: "b"}
	 var dest = {};
	 export(src, dest, "fieldA", null);
	 export(src, dest, "fieldB", null);
	 // dest 的值是 {fieldB: "b"}，因为 fieldA 为 null

format()function 
public function format(template:String, ... params):String

字符串格式化，有三种不同的用法 e1. format("{0} + {1} = {2}", 1, 2, 3); 得到 "1 + 2 = 3" e2. format("{0} + {1} = {2}", [2, 3, 5]); 得到 "2 + 3 = 5" e3. format("用户名：{username}, 注册日期：{created}", {username: "张三", created: "2010/9/12"}); 得到 "用户名：张三, 注册日期：2010/9/12"
